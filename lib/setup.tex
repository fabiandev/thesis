\RequirePackage[utf8]{inputenc}
\RequirePackage{etoolbox}

\newbool{footnotesbottom}

%% place footnotes at the very bottom
\makeatletter
  \newcommand{\footnotesbottom}{
    \ifbool{footnotesbottom}{}{\global\booltrue{footnotesbottom}}
  }
  
  %% append a path to the input paths
  \newcommand{\addlibpath}[1]{
    %% see https://tex.stackexchange.com/a/79060/122215
    \providecommand*{\input@path}{}
    \g@addto@macro\input@path{{#1}}
  }
\makeatother

%% automatically append lib/ to the input paths
\addlibpath{lib/}
